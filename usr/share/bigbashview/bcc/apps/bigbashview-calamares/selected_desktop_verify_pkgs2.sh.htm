#!/bin/bash ################################## # Author1: Bruno Goncalves (www.biglinux.com.br) # Author2: Rafael Ruscher (rruscher@gmail.com) # Date: 2022/02/28 # Modified:2022/08/05 # # Description: Calamares modified usage of BigLinux # # Licensed by GPL V2 or greater ################################## #Translation LANGUAGE=$LANG:en export TEXTDOMAINDIR="/usr/share/locale" export TEXTDOMAIN=calamares-biglinux NotPossibleInstall=$"Não é possível instalar.

" NotPossibleRemove=$"Não é possível remover.

" BACK_HISTORY=$"VOLTAR" CONTINUE_HISTORY=$"AVANÇAR" ERRO_MSG=$"ERRO!" SUCCESS_MSG=$"SUCESSO!" windowID="$(xprop -root '\t$0' _NET_ACTIVE_WINDOW | cut -f 2)" # Clean rm -f /tmp/start_calamares rm -f /tmp/biglinux-calamares-pkg-install.err rm -f /tmp/biglinux-calamares-pkg-remove.err pkgProblem="" echo "
" pacman -Si $(echo $install_packages | tr -d '\r') 2> /tmp/biglinux-calamares-pkg-install.err if [ "$?" != "0" ]; then pkgProblem="yes" fi echo "
" if [ "$remove_packages" != "" ]; then for package in $remove_packages; do if [ -z "$(grep "$package" /rootfs-pkgs.txt)" ]; then echo $package >> /tmp/biglinux-calamares-pkg-remove.err pkgProblem="yes" fi done fi if [ -e "/tmp/biglinux-calamares-pkg-install.err" ]; then msg_not_install="$NotPossibleInstall $(cat /tmp/biglinux-calamares-pkg-install.err | sed 's/\"//;s/\"/<\/b>/')" fi if [ -e "/tmp/biglinux-calamares-pkg-remove.err" ]; then msg_not_remove="$NotPossibleRemove $(cat /tmp/biglinux-calamares-pkg-remove.err)" fi cat << EOF $TITLE EOF # Get body tag with color light or not /usr/share/bigbashview/bcc/shell/getbgcolor.sh cat < EOF echo '
' echo $"O sistema está em modo" echo '' [ -d /sys/firmware/efi ] && echo ' UEFI' || echo ' BIOS (Legacy)' echo '' echo $"e com o kernel" echo '' echo "$(uname -r | cut -f1 -d-)" echo '' echo '
' cat << EOF
EOF if [ "$pkgProblem" = "yes" ]; then cat << EOF

$ERRO_MSG

EOF else cat << EOF

$SUCCESS_MSG

EOF fi echo '
' if [ "$pkgProblem" = "yes" ]; then echo "
$msg_not_install

$msg_not_remove
" else cat << EOF EOF fi echo '
' echo '' echo "" echo "" echo "" echo "" if [ "$pkgProblem" = "yes" ]; then cat << EOF $BACK_HISTORY EOF else cat << EOF $CONTINUE_HISTORY EOF fi cat << EOF
EOF
